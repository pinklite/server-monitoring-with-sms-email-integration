          export ENDPOINT={{ app.endpoint }};
          export DATACENTER=;

          export DATACENTER="aws ec2";
          export CLUSTER=webapi;
          curl -H 'X-Nodmonit-Auth: {{ app.auth }}' $ENDPOINT/install-agent | bash
